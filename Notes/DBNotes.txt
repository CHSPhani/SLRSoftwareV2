CREATE TABLE `slrv1`.`CommonParams` (
 `pID` INT NOT NULL,
 `pTitle` VARCHAR(150) NOT NULL,

 `pPublicationDate` VARCHAR(100) NOT NULL,

CREATE TABLE `slrv1`.`SubArea` (
